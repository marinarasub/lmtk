216 // help msg for usage of fx
217 std::string help_fx()
218 {
219 std::string all_fx;
220 all_fx.append("Use effects with --fx \"effect_name[#passes*](arg1, arg2,...)\" ...\n");
221 all_fx.append("Valid effects (* denotes optional argument):");
222
223 // find size to resize to
224 size_t max_name = 0, max_desc = 0, max_use = 0;
225 for (auto kv : fx_map)
226 {
227 size_t name_len = kv.first.length();
228 size_t desc_len = kv.second.description.length();
229 size_t use_len = kv.second.usage.length();
230 max_name = name_len > max_name ? name_len : max_name;
231 max_desc = desc_len > max_name ? desc_len : max_desc;
232 max_use = use_len > max_use ? use_len : max_use;
233 }
234 // format
235 for (auto kv : fx_map)
236 {
237 std::string name = kv.first;
238 std::string desc = kv.second.description;
239 std::string use = kv.second.usage;
240 name.resize(max_name, ' ');
241 desc.resize(max_desc, ' ');
242 use.resize(max_use, ' ');
243 all_fx.append("\n " + name);
244 all_fx.append(" " + desc);
245 all_fx.append(" args: " + use);
246 }
247 return all_fx;
248 }
